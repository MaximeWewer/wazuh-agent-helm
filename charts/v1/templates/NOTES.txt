COMMANDS
--------------------------------------------------------------------------------

# List pods:
kubectl get pods -n {{ .Release.Namespace }} -l app.kubernetes.io/name={{ include "wazuh-agent.name" . }}

# DaemonSet status:
kubectl get ds {{ include "wazuh-agent.fullname" . }} -n {{ .Release.Namespace }}

# View logs:
kubectl logs -n {{ .Release.Namespace }} -l app.kubernetes.io/name={{ include "wazuh-agent.name" . }} -c wazuh-agent --tail=20

# Agent status:
kubectl exec -n {{ .Release.Namespace }} -it <pod> -c wazuh-agent -- /var/ossec/bin/ossec-control status
